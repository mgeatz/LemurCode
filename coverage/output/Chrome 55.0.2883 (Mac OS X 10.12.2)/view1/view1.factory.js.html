angular.module('yesNoApp')
 
.factory('Charts', function(){
 
  return {
    sayHello: function(text){
      return text;
    },
    buildMap: function(statesArray){
      var map = AmCharts.makeChart( "chartdiv", {
        "type": "map",
        "theme": "light",
        "colorSteps": 10,
        "dataProvider": {
          "map": "usaLow",
          "areas": statesArray
        },
        "areasSettings": {
          "autoZoom": true
        },
        "valueLegend": {
          "right": 10,
          "minValue": "NO",
          "maxValue": "YES"
        },
        "export": {
          "enabled": true
        }
      });
      return map;
    }
  }
 
});
